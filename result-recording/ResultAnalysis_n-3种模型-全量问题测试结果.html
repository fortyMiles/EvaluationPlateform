%matplotlib inline

values = ['wrong', 'right', 'unsure']

def give_result_global_info(dataframe):
    dataframe.right.value_counts().plot(kind='pie')
    ratios = []
    for v in values:
        r = len(dataframe[dataframe.right == v]) / len(dataframe) 
        print('{}: {}%'.format(v, r * 100))
        ratios.append(v)
    assert len(ratios) == 3
    return ratios

def get_comb_val(c1, c2):
